root/ // Carpeta raíz del backend

├── node_modules/ // Dependencias instaladas (auto-generado)

│   │

│   │
│   ├── models/ // Modelos de datos (queries SQL, acceso a BD)
│   │
│   ├── routes/ // Definición de rutas y endpoints de la API
│   │
│   ├── services/ // Lógica de negocio (conecta controllers con models)
│   │
│   ├── test/ // Pruebas unitarias o de integración
│   │
│   ├── utils/ // Funciones auxiliares (helpers, formateos, etc.)
│   │
│   ├── validators/ // Validaciones de datos (Joi, express-validator, etc.)
│   │
│   └── app.js // Configuración principal de Express (middlewares, rutas)
